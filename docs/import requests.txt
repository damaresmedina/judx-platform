def extrair_andamento(url):
    try:
        response = requests.get(url)
        soup = BeautifulSoup(response.text, 'html.parser')
    
        andamento = soup.find_all('div', class_='andamento-processo') 
        if andamento:
            return ' | '.join([item.text.strip() for item in andamento])
        else:
            return 'Sem informações de andamento encontradas'
    
    except Exception as e:
        return f"Erro ao acessar o link: {e}"

resultados = []

for link in links:
    print(f"Extraindo dados do link: {link}")
    andamento = extrair_andamento(link)
    resultados.append({'link': link, 'andamento': andamento})
    time.sleep(1)  # Espera entre requisições para evitar sobrecarregar o servidor

# Converter os resultados em um DataFrame
df = pd.DataFrame(resultados)

# Salvar os resultados em um arquivo Excel
df.to_excel('andamentos_processos_STF.xlsx', index=False)
